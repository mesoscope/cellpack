[docs]    def FindFunctionByName(self, funcName):
        """find a function with a given name in a list of functions"""

        for f in self.callbacks:
            if f.__name__ == funcName:
                return f
        return None

[docs]    def CallCallbacks(self, *args, **kw):
        """call all callback fuctions"""

        results = []
        for func in self.callbacks:
            try:
                results.append(func(*args, **kw))

            except:
                print("ERROR ************************************************")
                traceback.print_exc()
                return "ERROR"

        return results

[docs]    def RemoveCallback(self, func):
        """Delete a callback fuction"""
        if type(func) == bytes:
            func = self.FindFunctionByName(func)
            if func is None:
                return "function %s not found" % func
        if func in self.callbacks:
            self.callbacks.remove(func)
        else:
            return "function %s not found" % func.__name__




[docs]class CallbackFunction:
    """Class to allow to specify arguments to a callback function"""

    def __init__(self, function, *args, **kw):
        self.function = function
        self.args = args
        self.kw = kw

    def __call__(self, *args, **kw):
        args = self.args + args
        kw.update(self.kw)
        return self.function(*args, **kw)
