import tkinter.filedialog, os

userhome = os.path.expanduser("~")



[docs]def fileOpenAsk(master, idir=None, ifile=None, types=None, title="Open"):
    if types == None:
        types = [("All files", "*")]
    file = tkinter.filedialog.askopenfilename(
        filetypes=types, initialdir=idir, initialfile=ifile, title=title
    )
    if file == "":
        file = None
    return file

        if folder:
