    v = kw.get(name)
    if name in list(kw.keys()):
        del kw[name]
    return v

    """get a vector and return it in a Numeric array"""

        if shape:
            if vector.shape == shape:
                return vector

[docs]def read_data(filename, function):
    """Read data from an ASCII file."""

    import string

    result = []
    assert type(filename) == bytes
    assert function in (None, float, int)
    print("reading", filename)
    f = open(filename, "r")
    while 1:
        line = f.readline()
        if not len(line):
            break
        if function:
            datum = list(map(function, string.split(line)))
            result.append(list(datum))
        else:
            result.append(line)
    f.close()
    if function == float:
        ar = Numeric.array(result).astype(viewerConst.FPRECISION)
    elif function == int:
        ar = Numeric.array(result).astype(viewerConst.IPRECISION)
    else:
        ar = result

    return ar

    """Write data to an ASCII file."""

    assert type(filename) == bytes
    import string

    f = open(filename, "w")

    def writedatum1(datum, f=f):
        line = str(datum)
        line = line[1:-1] + "\n"  # slice off the [ , ] characters
        f.write(line)

    def writedatum(datum, f=f):
        import string, numpy.oldnumeric as Numeric

        line = string.join(list(map(str, Numeric.array(datum))), " ") + "\n"
        f.write(line)

    list(map(writedatum, data))
    f.close()
